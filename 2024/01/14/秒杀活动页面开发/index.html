<script>
import {
    listActivitys,
    listAllProduct,
    addActivity,
    changeActivityStatus,
    delActivity,
    getActivity,
    updateActivity
} from '@/api/seckillactivity'
export default {
    name: 'User',
    data() {
        return {
            // 查询参数
            queryParams: {
                pageNum: 1,
                pageSize: 10,
                concertName: undefined,
                status: undefined
            },
            title: '',
            // 是否显示弹出层
            open: false,
            // 表单校验
            rules: {
                userName: [
                    { required: true, message: '用户名称不能为空', trigger: 'blur' },
                    {
                        min: 2,
                        max: 20,
                        message: '用户名称长度必须介于 2 和 20 之间',
                        trigger: 'blur'
                    }
                ],
                nickName: [
                    { required: true, message: '用户昵称不能为空', trigger: 'blur' }
                ],
                concertname: [{ required: true, message: '请选择商品名称', trigger: 'change' }],
                password: [
                    { required: true, message: '用户密码不能为空', trigger: 'blur' },
                    {
                        min: 5,
                        max: 20,
                        message: '用户密码长度必须介于 5 和 20 之间',
                        trigger: 'blur'
                    }
                ],
                email: [
                    {
                        type: 'email',
                        message: "'请输入正确的邮箱地址",
                        trigger: ['blur', 'change']
                    }
                ],
                phonenumber: [
                    {
                        pattern: /^1[3|4|5|6|7|8|9][0-9]\d{8}$/,
                        message: '请输入正确的手机号码',
                        trigger: 'blur'
                    }
                ]
            },
            // 角色选项
            concertOptions: [],
            // 显示搜索条件
            showSearch: true,
            // 遮罩层
            loading: true,
            // 用户表格数据
            concertList: null,
            // 总条数
            total: 0,
            // 选中数组
            ids: [],
            // 表单参数
            form: {},
            //更新还是新增
            type: true,
            // 商品描述
            description: '',
            // 弹出层显示
            dialogVisible: false
        }
    },
    watch: {},
    created() {
        this.getList()
    },
    methods: {
        /** 搜索按钮操作 */
        handleQuery() {
            this.queryParams.pageNum = 1
            this.getList()
        },
        // 多选框选中数据
        handleSelectionChange(selection) {
            this.ids = selection.map(item => item.id)
            this.single = selection.length !== 1
            this.multiple = !selection.length
        },
        // 取消按钮
        cancel() {
            this.open = false
            this.reset()
        },
        /** 查询用户列表 */
        getList() {
            this.loading = true
            listActivitys(this.queryParams).then((response) => {
                if (response.code == 4001)
                    this.$modal.msgError(response.msg)
                else {
                    this.concertList = response.data.rows
                    this.total = response.total
                }
                this.loading = false
            })
        },
        // 用户状态修改
        handleStatusChange(row) {
            const text = row.status === '0' ? '启用' : '停止'
            this.$modal.confirm('确认要' + text + '该秒杀活动吗？')
                .then(function () {
                    return changeActivityStatus(row.activityId, row.status)
                })
                .then((res) => {
                    if (res.code == 4001) {
                        this.$modal.msgError(res.msg)
                    } else {
                        this.$modal.msgSuccess(text + '成功')
                    }
                    this.getList();
                })
                .catch(function () {
                    row.status = row.status === '0' ? '1' : '0'
                })
        },
        /** 修改按钮操作 */
        handleUpdate(row) {
            this.type = false
            this.reset()
            const id = row.activityId
            listAllProduct().then((response) => {
              
                this.concertOptions = response.data
            })
            getActivity(id).then((response) => {
                if(response.code == 4001)
                this.$modal.msgError(response.msg)
                else{
                this.form = response.data
                console.log("this.form" + response.data)
                this.open = true
                this.title = '修改活动信息'
                }
                //this.form.password = response.password
            })
        },
        // 表单重置
        reset() {
            this.form = {
                concertId: undefined,
                concertName: undefined,
                startTime: undefined,
                endTime: undefined,
                description: undefined,
                status: '0'
            }
            this.resetForm('form')
        },
        /** 新增用户 */
        handleAdd() {
            this.reset()
            this.type = true
            listAllProduct().then((response) => {
                if(response.code == 4001)
                this.$modal.msgError(response.msg)
                else{
                this.concertOptions = response.data
                this.open = true
                this.title = '添加商品'
                }
            })
        },
        /** 商品描述 */
        handledes(row) {
            console.log(row)
            this.dialogVisible = true
            this.description = row.description
        },
        handleClose(done) {
            this.$confirm('确认关闭？')
                .then(_ => {
                    done();
                })
                .catch(_ => { });
        },
        handleConcertChange(selectedConcertname) {
            const selectedConcert = this.concertOptions.find(item => item.concertname === selectedConcertname);
            if (selectedConcert) {
                this.form.concertid = selectedConcert.concertid;
                this.form.concertname = selectedConcert.concertname;
            } else {
                // Handle the case where no match is found, reset values or provide a default behavior
                this.form.concertid = undefined;
                this.form.concertname = undefined;
            }

            this.$refs.form.validateField('concertname'); // 手动触发验证，确保concertname的验证状态正确更新
        },
        /** 删除按钮操作 */
        handleDelete(row) {
            const ids = row.activityId
            this.$modal.confirm('是否确认删除活动编号为"' + ids + '"的数据项？')
                .then(function () {
                    return delActivity(ids)
                })
                .then((res) => {
                    if(res.code == 4001)
                    this.$modal.msgError(res.msg)
                    else{
                    this.getList()
                    this.$modal.msgSuccess('删除成功')
                    }
                })
                .catch(() => { })
        },
        /** 提交按钮 */
        submitForm: function () {
            this.$refs['form'].validate((valid) => {
                if (valid) {
                    if (this.type == false) {
                        this.form.startTime = this.parseTime(this.form.startTime)
                        this.form.endTime = this.parseTime(this.form.endTime)
                        updateActivity(this.form).then((response) => {
                            this.$modal.msgSuccess('修改成功')
                            this.open = false
                            this.getList()
                        })
                    } else {
                        this.form.startTime = this.parseTime(this.form.startTime)
                        this.form.endTime = this.parseTime(this.form.endTime)
                        console.log(this.form.startTime)
                        addActivity(this.form).then((response) => {
                            this.$modal.msgSuccess('新增成功')
                            this.open = false
                            this.getList()
                        })
                    }
                }
            })
        }
    }
}
</script>

使用分页查询实现

@Service("seckillactivityService")
public class SeckillactivityServiceImpl extends ServiceImpl<SeckillactivityMapper, Seckillactivity> implements SeckillactivityService {

        /**
     * 查询所有秒杀活动
     *
     * @param seckillactivity 秒杀活动对象
     * @param pageSize  分页大小
     * @param pageNum  分页数
     * @return 查询结果
     */
    @Override
    public CommonResult<?> listAllSeckill(Seckillactivity seckillactivity, Integer pageSize, Integer pageNum) {
        LambdaQueryWrapper<Seckillactivity> queryWrapper = new LambdaQueryWrapper<>();
        queryWrapper.like(StringUtils.hasText(seckillactivity.getConcertName()), Seckillactivity::getConcertName, seckillactivity.getConcertName());
        queryWrapper.eq(StringUtils.hasText(seckillactivity.getStatus()), Seckillactivity::getStatus, seckillactivity.getStatus());
        Page<Seckillactivity> page = new Page<>();
        page.setCurrent(pageNum);
        page.setSize(pageSize);
        page(page, queryWrapper);

        //封装vo
        PageVo pageVo = new PageVo();
        pageVo.setRows(page.getRecords());
        pageVo.setTotal(page.getTotal());
        return CommonResult.success(pageVo);
    }

}

@RestController
@AuthToken
@RequestMapping("/seckillActivity")
public class SecKillActivityController {

    @Autowired
    private ConcertService concertService;
    @Autowired

    @Autowired
    private  PermissionUtil permissionUtil;
        /**
     * 获取音乐会名称列表
     *
     * @return CommonResult 音会名称列表
     */
    @GetMapping("/concertNameList")
    public CommonResult<?> concertNameList() {
        if(!permissionUtil.hasPermission("seckillactivity:add"))
            return CommonResult.error(4001, "无权限");
        return concertService.concertNameList();
    }


        /**
     * 获取所有秒杀活动列表
     *
     * @param seckillactivity 秒杀活动对象
     * @param pageSize 分页大小
     * @param pageNum 分页数
     * @return CommonResult<?> 返回结果对象
     */
    @GetMapping("/listAllSeckill")
    public CommonResult<?> listAllSeckill(Seckillactivity seckillactivity, Integer pageSize, Integer pageNum) {
        if(seckillactivity.getStatus() != null || seckillactivity.getConcertName() != null)
            if(!permissionUtil.hasPermission("seckillactivity:query"))
                return CommonResult.error(4001, "无权限");
        return seckillactivityService.listAllSeckill(seckillactivity, pageSize, pageNum);
    }


        /**
     * 添加秒杀活动
     *
     * @param seckillactivity 秒杀活动对象
     * @return CommonResult<?> 返回结果对象
     */
    @PostMapping()
    public CommonResult<?> addSeckillActivity(@RequestBody Seckillactivity seckillactivity) {

         return seckillactivityService.addSeckill(seckillactivity);
    }


        /**
     * 修改秒杀活动的状态
     * @param activityIdStatus 活动ID和状态的对象
     * @return 修改结果
     */
    @PutMapping("/changeStatus")
    public CommonResult<?> changeStatus(@RequestBody ActivityIdStatus activityIdStatus) {
        if(!permissionUtil.hasPermission("seckillactivity:edit"))
            return CommonResult.error(4001, "无权限");
        return seckillactivityService.changeStatus(activityIdStatus);
    }

        /**
     * 删除秒杀活动
     * @param id 秒杀活动ID
     * @return 删除结果
     */
    @DeleteMapping("/{id}")
    public CommonResult<?> delete(@PathVariable("id") Integer id) {
        if(!permissionUtil.hasPermission("seckillactivity:remove"))
            return CommonResult.error(4001, "无权限");
        return CommonResult.success(seckillactivityService.removeById(id));
    }

        /**
     * 根据活动ID获取秒杀活动信息
     *
     * @param id 活动ID
     * @return 秒杀活动信息
     */
    @GetMapping("/{id}")
    public CommonResult<?> getActivity(@PathVariable("id") Integer id) {
        if(!permissionUtil.hasPermission("seckillactivity:edit"))
            return CommonResult.error(4001, "无权限");
        return CommonResult.success(seckillactivityService.getById(id));
    }


        /**
     * 更新秒杀活动
     * 请求方式：PUT
     * 请求参数：秒杀活动对象
     * 响应结果：CommonResult
     */
    @PutMapping()
    public CommonResult<?> update(@RequestBody Seckillactivity seckillactivity) {
        return CommonResult.success(seckillactivityService.updateById(seckillactivity));
    }

}

@Component
public class PermissionUtil {

    private final RedisTemplate<String, String> redisTemplate;

    public PermissionUtil(RedisTemplate<String, String> redisTemplate) {
        this.redisTemplate = redisTemplate;
    }
        /**
     * 判断用户是否拥有指定权限
     * @param perm 需要判断的权限
     * @return 如果用户拥有指定权限，返回true；否则返回false
     */
    public boolean hasPermission(String perm) {
        // 从redis中获取权限列表
        String perms = redisTemplate.opsForValue().get("perms");
        // 将权限列表字符串转换为字符串列表
        List<String> permsList = JSONUtil.toList(perms, String.class);
        // 如果权限列表中包含指定权限，返回true；否则返回false
        return permsList.contains(perm);
    }

}
