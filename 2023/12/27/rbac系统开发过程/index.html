先留个坑，忙完期末写

本学期选修了一门互联网软件开发课程，收获颇丰。总结一下学期开发的项目，入门刚刚好。

一、项目介绍

本项目采用的技术栈为

后端：Springboot3、Redis、JWT、RabbitMQ、MybatisPlus…

前端：Vue2、element-ui…

数据库：MySQL

Java版本: jdk17

​ 该项目为基于RBAC权限设计的后台管理系统，在此基础上集成了一个演唱会秒杀系统。RBAC是什么？Role-Based Access Control：基于 角色 的权限控制 用户 -> 角色 -> 权限，简单来说就是在用户和权限之间加了一层角色，给角色赋予权限，再为用户赋予角色，这样实现了用户和权限之间的解耦。

主要记录后端开发以及测试部署过程，对于前端设计不作为重点。

项目地址:https://github.com/summerbird-1/SeckillSystem

二、项目准备

1、前端项目

前端使用的是Vue项目，https://github.com/summerbird-1/SeckillSystem该仓库web文件夹为前端工程。

打开终端 1、npm i 2、npm run serve

2、后端项目

后端项目使用的是Springboot3.17 + jdk17，用IDEA创建，目前没法直接创建jdk1.8和Springboot2.x的项目了。

以用户名生成token

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.JwtException;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;
import java.util.function.Function;

@Component
public class JwtTokenUtil {

    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.access_token.expiration}")
    private Long accessTokenExpiration;

    @Value("${jwt.refresh_token.expiration}")
    private Long refreshTokenExpiration;

    public String generateAccessToken(String username) {
        Map<String, Object> claims = new HashMap<>();
        return doGenerateToken(claims, username, accessTokenExpiration);
    }

    public String generateRefreshToken(String username) {
        Map<String, Object> claims = new HashMap<>();
        return doGenerateToken(claims, username, refreshTokenExpiration);
    }

    private String doGenerateToken(Map<String, Object> claims, String username, Long expiration) {
        return Jwts.builder()
                .setClaims(claims)
                .setSubject(username)
                .setIssuedAt(new Date(System.currentTimeMillis()))
                .setExpiration(new Date(System.currentTimeMillis() + expiration * 1000))
                .signWith(SignatureAlgorithm.HS256,secret)
                .compact();
    }

    public String getUsernameFromToken(String token) {
                return getClaimFromToken(token, Claims::getSubject);
    }

    public Date getExpirationDateFromToken(String token) {
        return getClaimFromToken(token, Claims::getExpiration);
    }

    public <T> T getClaimFromToken(String token, Function<Claims, T> claimsResolver) {
        Claims claims = getAllClaimsFromToken(token);
        return claimsResolver.apply(claims);
    }

    private Claims getAllClaimsFromToken(String token) {
        return Jwts.parser().setSigningKey(secret).parseClaimsJws(token).getBody();
    }

    private Boolean isTokenExpired(String token) {
        Date expiration = getExpirationDateFromToken(token);
        return expiration.before(new Date());
    }

    public Boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return !isTokenExpired(token);
        } catch (JwtException | IllegalArgumentException e) {
            return false;
        }
    }
}

@Autowired
private JwtTokenUtil jwtTokenUtil;
@Autowired
private RedisTemplate<String, String> redisTemplate;
@Autowired
private Environment environment;
/**
     * 用户登录
     *
     * @param loginRequest 登录请求
     * @return 登录结果
     */
    @Override
    public CommonResult<?> login(LoginRequest loginRequest) {
        // 创建查询条件
        LambdaQueryWrapper<User> queryWrapper = new LambdaQueryWrapper<>();
        queryWrapper.eq(User::getUsername, loginRequest.getUsername());
        // 根据查询条件获取用户
        User user = getOne(queryWrapper);
        // 如果用户为空，返回错误结果
        if (Objects.isNull(user)) {
            return CommonResult.error(500, "用户名或密码错误");
        }
        // 检查密码是否正确
        if (!user.getPassword().equals(loginRequest.getPassword())) {
            return CommonResult.error(500, "用户名或密码错误");
        }
        // 获取用户名
        String username = user.getUsername();
        // 生成访问令牌
        String accessToken = jwtTokenUtil.generateAccessToken(username);
        // 生成刷新令牌
        String refreshToken = jwtTokenUtil.generateRefreshToken(username);
        // 获取访问令牌和刷新令牌的过期时间
        Long accessTokenTTL = Long.valueOf(Objects.requireNonNull(environment.getProperty("jwt.access_token.expiration")));
        Long refreshTokenTTL = Long.valueOf(Objects.requireNonNull(environment.getProperty("jwt.refresh_token.expiration")));
        // 将访问令牌和刷新令牌存入Redis
        redisTemplate.opsForValue().set(accessToken, "accessToken", accessTokenTTL, TimeUnit.SECONDS);
        redisTemplate.opsForValue().set(refreshToken, "refreshToken", refreshTokenTTL, TimeUnit.SECONDS);
        // 创建TokenResponse对象
        TokenResponse tokenResponse = new TokenResponse(accessToken, refreshToken);
        // 返回成功结果
        return CommonResult.success(tokenResponse);
    }

        result.data = data;
        result.msg = "操作成功";
        return result;
    }

    /**
     * 列出用户页面的用户信息
     *
     * @param user 当前用户对象
     * @param pageNum 当前页码
     * @param pageSize 每页用户数量
     * @return 返回包含用户信息的分页对象
     */
    @Override
    public PageVo list(User user, Integer pageNum, Integer pageSize) {
        LambdaQueryWrapper<User> queryWrapper = new LambdaQueryWrapper<>();
        queryWrapper.like(StringUtils.hasText(user.getUsername()), User::getUsername, user.getUsername());
        queryWrapper.eq(StringUtils.hasText(user.getPhone()), User::getPhone, user.getPhone());
        queryWrapper.eq(StringUtils.hasText(user.getStatus()), User::getStatus, user.getStatus());
        Page<User> page = new Page<>();
        page.setCurrent(pageNum);
        page.setSize(pageSize);
        page(page, queryWrapper);

        List<User> userList = page.getRecords();
        List<UserVo> userVos = BeanCopyUtils.copyBeanList(userList, UserVo.class);
        PageVo pageVo = new PageVo();
        pageVo.setRows(userVos);
        pageVo.setTotal(page.getTotal());
        return pageVo;
    }

public class BeanCopyUtils {
    private BeanCopyUtils(){

    }
    public static <V> V copyBean(Object source,Class<V> clazz) {

        V result = null;
        try {
            //创建目标对象
            result = clazz.newInstance();
            //实现属性copy
            BeanUtils.copyProperties(source, result);


        } catch (Exception e) {
            e.printStackTrace();
        }
        //返回结果

        return result;
    }
    public static <O,V> List<V> copyBeanList(List<O> list, Class<V> clazz){
        return list.stream()
                .map(o -> copyBean(o, clazz))
                .collect(Collectors.toList());
    }
}
